<script>
function startSite(){
  document.getElementById("start").classList.add("hidden");
  document.getElementById("main").classList.remove("hidden");
}

const reasons=[
"Your laugh always makes my day better.",
"You make normal days feel special.",
"I feel comfortable being myself around you.",
"You care about people in a really genuine way.",
"You make my life more fun just by being in it."
];

function clearExtras(){
  document.getElementById("starLayer").innerHTML="";
  document.getElementById("starLayer").classList.add("hidden");
}

function showReasons(){
  clearExtras();
  const random=reasons[Math.floor(Math.random()*reasons.length)];
  document.getElementById("contentBox").innerText=random;
}

function showMemory(){
  clearExtras();

  document.getElementById("contentBox").innerText="One of my favorite memories that I still think about is November 18th. We were sitting under the stars, wrapped up in blankets, just talking. It felt like time slowed down. We both knew what we wanted to say, but neither of us could find the words. That moment still means a lot to me.";

  createStars();
}

function showSong(){
  clearExtras();

  document.getElementById("contentBox").innerHTML="<strong>The Line – Twenty One Pilots</strong><br><br>This song reminds me of you because we were listening to it on October 18th while driving to the mall in the rain. Everything about that moment just felt perfect and peaceful, and it's one of those memories I still think about.<br><br><a href='https://music.apple.com/us/search?term=the%20line%20twenty%20one%20pilots' target='_blank'>Open in Apple Music</a>";
}

function showTimeTogether(){
  clearExtras();

  const startDate=new Date("2025-10-18");
  const today=new Date();

  const diffTime=today-startDate;
  const days=Math.floor(diffTime/(1000*60*60*24));

  document.getElementById("contentBox").innerHTML=`
  <strong>Time We've Been Together</strong>
  <div class='counter'>${days} days</div>
  <p>And every single one of them has meant a lot to me.</p>
  <button onclick="revealSecret()">Click for a secret message</button>
  <div id='secretMsg' class='secret'></div>
  `;
}

function revealSecret(){
  document.getElementById("secretMsg").innerText="I'm really glad I get to spend this time with you. Happy 5 months ❤️";
}

function createStars(){
  const layer=document.getElementById("starLayer");
  layer.classList.remove("hidden");
  layer.classList.add("stars");

  for(let i=0;i<120;i++){
    const star=document.createElement("div");
    star.classList.add("star");
    star.style.left=Math.random()*100+"%";
    star.style.top=Math.random()*100+"%";
    star.style.animationDuration=(Math.random()*2+1)+"s";
    layer.appendChild(star);
  }
}
